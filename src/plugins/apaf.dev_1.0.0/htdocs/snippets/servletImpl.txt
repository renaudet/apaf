/*
 * ctx: {
 	"user": <connected user>,
 	"runtime": <npa runtime>,
 	"require": <Node.js 'require' function>,
 	"httpRequest": <original Express request>
   }
   then: function(err,result)
 */
myFunction = function(params,ctx,then){
	let paths = ctx.httpRequest.params.alias;
	//let servletAlias = paths[0];
	let version = '1.0';
	if(paths.length>1){
		version = paths[1];
	}
	then(null,{"status": "ok","version": version});
}

servlet.endpoint = myFunction;
